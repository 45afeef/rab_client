Use the following code to generate API client. 
This creates Dart Client

docker run --rm `
  -v "${PWD}:/local" `
  openapitools/openapi-generator-cli generate `
  -i https://rab-fastapi-predev-135325225027.asia-south1.run.app/api/v1/openapi.json `
  -g dart-dio `
  -o /local/dart_sdk `
  --additional-properties=pubName=rab_dio,pubVersion=1.0.0,nullSafety=true


docker run --rm `
  -v "${PWD}:/local" `
  openapitools/openapi-generator-cli generate `
  -i http://host.docker.internal:8000/api/v1/openapi.json `
  -g dart-dio `
  -o /local/dart_sdk `
  --additional-properties=pubName=rab_dio,pubVersion=1.0.0,nullSafety=true
